```python
def calculate_compliance_score(framework):
    """Calculate real-time compliance score"""
    # Get all controls for framework
    controls = get_framework_controls(framework)

    scores = {
        'total_controls': len(controls),
        'passing': 0,
        'failing': 0,
        'not_tested': 0,
        'manual_review': 0
    }

    for control in controls:
        status = get_control_status(control['id'])

        if status == 'PASS':
            scores['passing'] += 1
        elif status == 'FAIL':
            scores['failing'] += 1
        elif status == 'NOT_TESTED':
            scores['not_tested'] += 1
        elif status == 'MANUAL_REVIEW':
            scores['manual_review'] += 1

    # Calculate percentage
    tested = scores['passing'] + scores['failing']
    if tested > 0:
        scores['compliance_percentage'] = (scores['passing'] / tested) * 100
    else:
        scores['compliance_percentage'] = 0

    # Risk score
    scores['risk_score'] = calculate_risk_score(scores)

    return scores


def calculate_risk_score(scores):
    """Calculate overall risk score (0-100)"""
    # Failing controls = higher risk
    failing_weight = scores['failing'] * 10

    # Not tested = medium risk (unknown state)
    untested_weight = scores['not_tested'] * 5

    # Manual review = low risk (being addressed)
    manual_weight = scores['manual_review'] * 2

    total_risk = failing_weight + untested_weight + manual_weight
    max_risk = scores['total_controls'] * 10

    # Invert (100 = no risk, 0 = maximum risk)
    risk_score = 100 - (total_risk / max_risk * 100) if max_risk > 0 else 100

    return round(risk_score, 1)
```

```python
def analyze_compliance_trends(framework, days=90):
    """Analyze compliance trends and forecast"""
    # Get historical scores
    historical_scores = get_historical_scores(framework, days=days)

    # Calculate trend
    import numpy as np
    from scipy import stats

    dates = [s['date'] for s in historical_scores]
    scores = [s['score'] for s in historical_scores]

    # Linear regression
    x = np.arange(len(dates))
    slope, intercept, r_value, p_value, std_err = stats.linregress(x, scores)

    # Forecast next 30 days
    forecast_days = 30
    forecast_x = np.arange(len(dates), len(dates) + forecast_days)
    forecast_scores = slope * forecast_x + intercept

    # Determine trend
    if slope > 0.1:
        trend = 'IMPROVING'
    elif slope < -0.1:
        trend = 'DEGRADING'
    else:
        trend = 'STABLE'

    return {
        'current_score': scores[-1],
        'trend': trend,
        'slope': slope,
        '30_day_forecast': forecast_scores[-1],
        'confidence': r_value ** 2,
        'recommendation': generate_recommendation(trend, scores[-1])
    }
```
